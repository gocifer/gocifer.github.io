# gentoo

#!/bin/bash
mkdir -p /mnt/fedora/boot/efi
umount /dev/sdb1
mount /dev/mapper/Gentoo-main /mnt/gentoo
mount /dev/mapper/Gentoo-boot /mnt/gentoo/boot
mount /dev/sdb1 /mnt/gentoo/boot/efi
mount -t proc /proc /mnt/gentoo/proc
mount --rbind /sys /mnt/gentoo/sys
mount --make-rslave /mnt/gentoo/sys
mount --rbind /dev /mnt/gentoo/dev
mount --make-rslave /mnt/gentoo/dev
cp /mnt/gentoo/etc/resolv.conf /mnt/gentoo/etc/resolv.conf.bak
cp /etc/resolv.conf /mnt/gentoo/etc/ 
chroot /mnt/gentoo /bin/bash
