# gentoo

#!/bin/bash
mkdir -p /mnt/fedora/boot/efi
umount /dev/sdb1
mount /dev/mapper/Gentoo-main /mnt/gentoo
mount /dev/mapper/Gentoo-boot /mnt/gentoo/boot
mount /dev/sdb1 /mnt/gentoo/boot/efi
mount -t proc /proc /mnt/gentoo/proc
mount --rbind /sys /mnt/gentoo/sys
mount --make-rslave /mnt/gentoo/sys
mount --rbind /dev /mnt/gentoo/dev
mount --make-rslave /mnt/gentoo/dev
cp /mnt/gentoo/etc/resolv.conf /mnt/gentoo/etc/resolv.conf.bak
cp /etc/resolv.conf /mnt/gentoo/etc/ 
chroot /mnt/gentoo /bin/bash

# fedora

#!/bin/bash

mkdir -p /mnt/fedora/boot/efi
umount /dev/sdb1
umount /dev/mapper/Fedora-root
mount /dev/sdb1 /mnt/fedora/boot/efi
mount /dev/mapper/Fedora-root /mnt/fedora
mount /dev/mapper/Fedora-boot /mnt/fedora/boot
mount /dev/mapper/Fedora-home /mnt/fedora/home
mount /dev/mapper/Fedora-var /mnt/fedora/var
mount -t proc /proc /mnt/fedora/proc
mount --rbind /sys /mnt/fedora/sys
mount --make-rslave /mnt/fedora/sys
mount --rbind /dev /mnt/fedora/dev
mount --make-rslave /mnt/fedora/dev
mv /mnt/fedora/etc/resolv.conf /mnt/fedora/etc/resolv.conf.bak
cp /etc/resolv.conf /mnt/fedora/etc/ 
chroot /mnt/fedora /bin/bash
